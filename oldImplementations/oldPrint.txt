
void print(char *string, int color)
{
	char *vidmem = (char *)0xb8000;

	register int chars;
	register int charsInStr = 0;
	register int i;
	register int j;

	for (chars = 0; string[chars] != '\0'; chars++);

	int charsValueForColorLoop = chars * 2;
	int charsValueForCharPlacement = chars * 2;

	for (j = 0; j < charsValueForColorLoop; j++)
	{
		j += 1;
		vidmem[j] = color;
	}

	for (i = 0; i < charsValueForCharPlacement; i++)
	{
		vidmem[i] = string[charsInStr];
		i += 1;
		charsInStr += 1;
	}
}
